from fastapi import FastAPI
from fastapi.responses import JSONResponse
from fastapi.middleware.cors import CORSMiddleware
import pandas as pd

app = FastAPI()

# Enable CORS
origins = [
    "http://localhost:3000",  # Replace with the URL of your React app
    # Add more origins if needed
]

app.add_middleware(
    CORSMiddleware,
    allow_origins=origins,
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# Load CSV data into a DataFrame
df = pd.read_csv('updated_drone_data.csv')

@app.get("/")
async def get_drone_data():
    data = df.to_dict(orient='records')
    return JSONResponse(data)

@app.get("/{index}")
async def get_drone_data_by_index(index: int):
    if index < 0 or index >= len(df):
        return JSONResponse({"error": "Index out of bounds"}, status_code=404)
    row = df.iloc[index].to_dict()
    return JSONResponse(row)

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
